Basic usage

ProgressBar progressBar = new ProgressBar("progressBar");
progressBar.setMax(200);		
progressBar.setActual(25);
$r.render("base") 
progressBar.setWidth(300);
progressBar.setUnitLabel("MB");
progressBar.setLabelAlignment(LabelAlignment.right);
progressBar.setMax(125);		
progressBar.setActual(40);
$r.render("base2")

Render Styles

progressBar.setRenderStyle(RenderStyle.horizontal);
$r.render("hor") 
progressBar.setRenderStyle(RenderStyle.radial);
$r.render("rad") 
progressBar.setRenderStyle(RenderStyle.pie);
$r.render("pie")
